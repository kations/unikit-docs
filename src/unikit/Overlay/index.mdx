import { useState } from 'react'
import { Playground, Props } from 'docz'
import {Flex, Button, Overlay} from '../index'

# Overlay

### Usage

```bash
import { Overlay } from 'unikit'
```

<Playground>
    {() => {
        const [overlay, setOverlay] = useState(false);
        return (
     
            <Flex bg="background" p={30} content="center">
                <Button onPress={() => setOverlay(true)}>show overlay</Button>
                <Overlay scrollable p={30} position="center" contentProps={{ p: 20, maxWidth: 500, width: "90%", bg: "surface" }} visible={overlay} onClose={() => setOverlay(false)}>
                    <Button onPress={() => setOverlay(false)}>close overlay</Button>
                </Overlay>
            </Flex>
    
        )
    }}
    
</Playground>
